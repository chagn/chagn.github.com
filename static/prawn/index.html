頁碼 number_pages
# 頁碼的字串, 會自動帶入<page> & <total>
string = "Page:  <page>  of  <total> "  
options = {  
  :at => [bounds.right - 150, bounds.top - 55], # 出現的位置, 不確定能不能複數個
  :width => 150, # box的寬度
  :align => :right,
  :page_filter => lambda{ |pg| pg != 1 }, # 哪幾頁不顯示
  :start_count_at => 1, # 開始的數字
  :total_pages => (page_count - 1), # page_count => 總頁數,  扣掉不要的就是你要計算的頁面數, 這裡扣掉封面，所以 -1
  :color => "000000"
}
number_pages string, options
